\def\imp{\to}
\def\land{\mathbin\&}

\documentstyle[proof]{article}

% for LaTeX 2e naitive mode.

\begin{document}
\section{Examples of proof.sty}

\verb|\infer| draws beautiful proof figures easily:
\infer{E}{
                       A
                       &
                       \infer{D}{B & C}
               }

\noindent(1)

\infer{A_1}{
        \infer=[\imp E_1]{B_1}{
                \frac{\Pi}{2}
                &
                \infer[\wedge E_1]{B_21}{B_21 \wedge B22} 
                & 
                D1 \wedge D2
        }
        &
        \infer={C_1}{C_11 \wedge C12}
}


\noindent(2)

\infer={A}{
        \infer=[\imp E]{B}{
                \frac{\Pi}{2}
                &
                \infer*[\wedge E_1]{B21}{B21 \wedge B22} 
                & 
                D1 \wedge D2
        }
        &
        \deduce{C}{C11 \wedge C12}
}


\noindent (3)

\infer=[RI]{A \imp B}{
        \infer*{B}{[A]}
}


\noindent (4)

\infer=[(1)]{A}{
        \infer{B}{
                B11\land B12\land B13
                &
                B21\land B22\land B23
        }
        &
        C
}


\noindent (5)

\infer[(\wedge I)]{A \wedge B}{A & B}
\qquad
\infer[(\wedge E_1)]{A}{A \wedge B}
\qquad
\infer[(\wedge E_r)]{B}{A \wedge B}
