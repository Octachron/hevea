\documentstyle[proof]{article}

\begin{document}
\section*{Examples of proof.sty}

\verb|\infer| draws beautiful proof figures easily:

\noindent (1)
$$
\infer{A}{
	\infer{B}{
		B11\land B12
		&
		B21\land B22
	}
	&
	C
}
$$

\noindent (2)
$$
\infer{A \imp B}{
	\infer{B}{[A]}
}
$$
